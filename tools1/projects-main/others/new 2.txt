app.post("/createdatasource", (req, res) => {
  const datasource_name = req.body.datasource_name;
  const datasource_type = req.body.datasource_type;
  const active_status = req.body.active_status;
  const expiry_date = req.body.expiry_date;

  db.run(
    "CREATE TABLE IF NOT EXISTS DataSource(id INTEGER PRIMARY KEY AUTOINCREMENT, datasource_name TEXT, datasource_type TEXT, active_status TEXT, expiry_date TEXT )",
    (err, result) => {
      if (err) {
        console.log(err);
        res.send("Failure");
      } else {
        db.run(
          "INSERT INTO DataSource (datasource_name, datasource_type, active_status, expiry_date) VALUES (?,?,?,?)",
          [datasource_name, datasource_type, active_status, expiry_date],
          (err, result) => {
            if (err) {
              console.log(err);
              res.send("Failure");
            } else {
              res.send("Success");
            }
          }
        );
      }
    }
  );
});
